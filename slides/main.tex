\begin{frame}{Законы масштабирования и достаточный размер выборки}
    Исследуется связь между архитектурой параметрической модели, локальной геометрией оптимизационной поверхности и объемом данных для устойчивого обучения.
    \begin{block}{Проблема}
        Эмпирические законы масштабирования для современных моделей хорошо наблюдаются на практике, однако не имеют общего теоретического объяснения. При этом множество оптимальных параметров зависит от конечной обучающей выборки.
    \end{block}
    \begin{block}{Требуется}
        Построить теоретическую схему, связывающую архитектуру модели, локальную кривизну поверхности функции потерь и достаточный размер обучающей выборки.
    \end{block}
    \begin{block}{Метод решения}
        Предлагается анализировать \alert{\textbf{стабилизацию ландшафта функции потерь}} при увеличении объема выборки. Для линейных моделей используются апостериорные распределения параметров, для нейросетей --- локальная геометрия поверхности и спектральные свойства матрицы Гессе.
    \end{block}
\end{frame}

\begin{frame}{Разложение Гаусса"--~Ньютона матрицы Гессе}
\begin{columns}[T]
    \begin{column}{0.56\textwidth}
        \begin{block}{Эмпирическая функция потерь}
            \vspace{-0.6em}
            \[
                \mathfrak D_m=\{(\mathbf x_i,\mathbf y_i)\}_{i=1}^m,
                \quad \mathbf{x} \in \mathbb{R}^{n},
                \quad \mathbf{y} \in \mathbb{R}^{K}
            \]
            \[
                \mathcal L_m(\mathbf w)=\frac1m\sum_{i=1}^m \ell(f_{\mathbf w}(\mathbf x_i),\mathbf y_i)
            \]
            \vspace{-1.8em}
        \end{block}
        \begin{block}{Матрица Гессе}
            \vspace{-1.8em}
            \[
                \mathbf H^{(m)}(\mathbf w)
                =
                \nabla_{\mathbf w}^2 \mathcal L_m(\mathbf w)
                =
                \frac1m\sum_{i=1}^m \mathbf H_i(\mathbf w)
                \in \mathbb{R}^{N \times N}
            \]
            \vspace{-1.8em}
        \end{block}
        \begin{block}{Обозначения на одном объекте}
            \vspace{-1.8em}
            \[
                \mathbf z=f_{\mathbf w}(\mathbf x) \in \mathbb{R}^{K},\quad
                \mathbf J(\mathbf w)=\frac{\partial \mathbf z}{\partial \mathbf w} \in \mathbb{R}^{K \times N},
            \]
            \[
                \mathbf A(\mathbf w)=\nabla_{\mathbf z}^2 \ell(\mathbf z,\mathbf y) \in \mathbb{R}^{K \times K}
            \]
            \vspace{-1.2em}
        \end{block}
    \end{column}
    \begin{column}{0.44\textwidth}
        \begin{block}{Разложение Гаусса"--~Ньютона}
            \vspace{-0.6em}
            \[
                \mathbf H(\mathbf w)=\mathbf G(\mathbf w)+\mathbf R(\mathbf w)
            \]
            \[
                \mathbf G(\mathbf w)=\mathbf J(\mathbf w)^\top \mathbf A(\mathbf w)\mathbf J(\mathbf w)
            \]
            \[
                \mathbf R(\mathbf w)=\sum_{k=1}^{K}
                \frac{\partial \ell(\mathbf z,\mathbf y)}{\partial z_k}\,
                \nabla_{\mathbf w}^2 z_k
            \]
            \vspace{-1.2em}
        \end{block}
        \begin{block}{Ключевая идея}
            В окрестности минимума $\|\mathbf G(\mathbf w)\|_2 \approx \|\mathbf H(\mathbf w)\|_2$, поэтому далее локальная кривизна анализируется через \alert{\textbf{гаусс"--~ньютоновскую компоненту}} $\|\mathbf G(\mathbf w)\|_2$.
        \end{block}
    \end{column}
\end{columns}
\end{frame}

\begin{frame}{Блочная факторизация гаусс"--~ньютоновская компоненты $\mathbf{G}(\mathbf{w})$}
    \vspace{-0.6em}
    \begin{figure}[ht]
        \centering
        \resizebox{0.8\linewidth}{!}{\input{figs/gn_factorization.tikz}}
    \end{figure}
    \vspace{-1.0em}
    \begin{block}{Лемма (Киселев, 2026)}
        Для нейронной сети $f_{\mathbf{w}}$ с архитектурой $\mathbf{z}^{(p)} = T_{\mathbf w^{(p)}}^{(p)}(\mathbf{x}^{(p)})$, $\mathbf{x}^{(p+1)} = \sigma^{(p)}(\mathbf{z}^{(p)})$, где операторы $T_{\mathbf w^{(p)}}^{(p)}$ линейны по входу, каждый блок $\mathbf{G}^{(p,q)}(\mathbf{w})$ выражается через параметрические якобианы слоев~$\mathbf{P}^{(p)}(\mathbf{w}) = \partial \mathbf{z}^{(p)} / \partial \mathbf{w}^{(p)}$ и матрицы распространения возмущения~$\mathbf{S}^{(p)}(\mathbf{w}) = \partial \mathbf{z} / \partial \mathbf{z}^{(p)}$ по формуле $\mathbf G^{(p,q)}(\mathbf w) = \mathbf P^{(p)\top}\mathbf S^{(p)\top}\mathbf A(\mathbf w)\mathbf S^{(q)}\mathbf P^{(q)}$.
    \end{block}
\end{frame}

\begin{frame}{Критерий сходимости поверхности функции потерь}
\begin{columns}[T]
    \begin{column}{0.57\textwidth}
        \hspace*{0.6em}
        \begin{minipage}{\linewidth}
            Критерий $p$-сходимости ландшафта функции потерь при добавлении одного объекта в выборку~$\mathfrak{D}_k$:
            \begin{equation*}
                \Delta_{p}(k+1) = \int | \mathcal{L}_{k+1}(\mathbf{w}) - \mathcal{L}_{k}(\mathbf{w}) |^{p} q(\mathbf{w}) d\mathbf{w},
            \end{equation*}
            где функция $q: \mathbb{R}^{N} \to \mathbb{R}_{+}$ задает предпочтение по выбору точек в пространстве параметров модели для оценки сходимости.
        \end{minipage}
    \end{column}
    \hspace{0.2em}
    \begin{column}{0.4\textwidth}
        \begin{figure}
            \centering
            \includegraphics[width=\linewidth]{criterion_general.png}
        \end{figure}
    \end{column}
\end{columns}
\vspace{1em}
\begin{enumerate}
    \item Без ограничения общности, $\int q(\mathbf{w}) d\mathbf{w} = 1$, т.е. $q(\mathbf{w})$ суть плотность распределения;
    \item Нас интересует локальная геометрия в окрестности минимума $\to$ определяем $q(\mathbf{w})$ с основной массой возле минимума.
\end{enumerate}
\end{frame}

\begin{frame}{Абсолютный одноточечный критерий сходимости}
\begin{columns}[T]
    \begin{column}{0.57\textwidth}
        \vspace{2.0em}
        \hspace*{0.1em}
        \begin{minipage}{\linewidth}
            Абсолютный одноточечный критерий сходимости:
            \begin{equation*}
                \Delta_{1}(k+1) = | \mathcal{L}_{k+1}(\mathbf{w}) - \mathcal{L}_{k}(\mathbf{w}) |,
            \end{equation*}
            где $\mathbf{w} \in \mathcal{U}_{R}(\mathbf{w}_k^*)$ "--- точка из окрестности радиуса $R>0$ минимума $\mathbf{w}_k^{*}$, т.е. $p=1$ и $q(\tilde{\mathbf{w}}) = \delta(\tilde{\mathbf{w}} - \mathbf{w})$.
        \end{minipage}
    \end{column}
    \hspace{0.2em}
    \begin{column}{0.4\textwidth}
        \begin{figure}
            \centering
            \includegraphics[width=\linewidth]{criterion_abs.png}
        \end{figure}
    \end{column}
\end{columns}
\begin{block}{Теорема (Киселев, 2024)}
    Пусть в $\mathcal{U}_{R}(\mathbf{w}_k^*)$ справедлива квадратичная аппроксимация, $\| \mathbf{g}_{k+1}(\mathbf{w}_k^*) \|_2 \leqslant M_{\mathbf{g}}$ и для всех $i = 1, \ldots, k+1$ верно $| \ell_{i}(\mathbf{w}_k^*) | \leqslant M_{\ell}$, $\| \mathbf{H}_{i}(\mathbf{w}_k^*) \|_2 \leqslant M_{\mathbf{H}}$, тогда
    \begin{equation*}
        \Delta_{1}(k+1) \leqslant \frac{2 M_{\ell} + M_{\mathbf{g}} R + M_{\mathbf{H}} R^2}{k + 1} = \mathcal{O}(k^{-1}).
    \end{equation*}
\end{block}
\end{frame}

\begin{frame}{Среднеквадратичный критерий сходимости}
\begin{columns}[T]
    \begin{column}{0.57\textwidth}
        \hspace*{0.1em}
        \begin{minipage}{\linewidth}
            Среднеквадратичный критерий сходимости:
            \vspace{-0.2em}
            \begin{equation*}
                \Delta_{2}(k+1) = \mathbb{E}_{q(\mathbf{w})} \left[ \big( \mathcal{L}_{k+1}(\mathbf{w}) - \mathcal{L}_{k}(\mathbf{w}) \big)^2 \right],
            \end{equation*}
            \vspace{-1.2em}
            
            т.е. $p=2$, для теоретического анализа удобно выбрать $q(\mathbf{w}) = \mathcal{N}(\mathbf{w}_k^*, \sigma^2 \mathbf{I}_N)$, а на практике вычисляется методом Монте"--~Карло через $\mathbf{w}_s \sim q(\mathbf{w})$:
            \vspace{-1.0em}
            \begin{equation*}
                \hat{\Delta}_{2}(k+1) = \frac{1}{S} \sum_{s=1}^{S} \big( \mathcal{L}_{k+1}(\mathbf{w}_s) - \mathcal{L}_{k}(\mathbf{w}_s) \big)^2.
            \end{equation*}
        \end{minipage}
    \end{column}
    \hspace{0.2em}
    \begin{column}{0.4\textwidth}
        \begin{figure}
            \centering
            \includegraphics[width=\linewidth]{criterion_squared.png}
        \end{figure}
    \end{column}
\end{columns}
\begin{block}{Теорема (Киселев, 2025)}
    Пусть в $\mathcal{U}_{R}(\mathbf{w}_k^*)$ справедлива квадратичная аппроксимация, $\| \mathbf{g}_{k+1}(\mathbf{w}_k^*) \|_2 \leqslant M_{\mathbf{g}}$ и для всех $i = 1, \ldots, k+1$ верно $| \ell_{i}(\mathbf{w}_k^*) | \leqslant M_{\ell}$, $\| \mathbf{H}_{i}(\mathbf{w}_k^*) \|_2 \leqslant M_{\mathbf{H}}$, тогда

        \Delta_{2}(k+1) \leqslant \frac{12 M_{\ell}^2 + 3 \sigma^2 M_{\mathbf{g}}^2 + 3 \sigma^4 (N^2 + 2N) M_{\mathbf{H}}}{(k + 1)^2} = \mathcal{O}(k^{-2}).
    \end{equation*}
\end{block}
\end{frame}

\begin{frame}{Достаточный размер выборки и законы масштабирования}
\begin{columns}[T]
    \begin{column}{0.4\textwidth}
        \begin{block}{Достаточный размер выборки}
            \vspace{-1.6em}
            \begin{equation*}
                \Delta(m) \in \left\{
                \Delta_1(m),\,
                \Delta_2(m),\,
                \Delta_2^{(D)}(m)
                \right\}
            \end{equation*}
            \begin{equation*}
                \forall m \geqslant m^*
                \qquad
                \Delta(m)\leqslant \varepsilon
            \end{equation*}
        \end{block}
        \vspace{-1.0em}
        \begin{block}{Архитектурно-зависимая локальная кривизна}
            \vspace{-2.0em}
            \[
                M_{\mathbf G}(\mathcal A)
                =
                \sup_{\mathbf w \in \mathcal U_R(\mathbf w^*)}
                \sup_i
                \|\mathbf G_i(\mathbf w)\|_2
            \]
            \[
                \|\mathbf H_i(\mathbf w)\|_2
                \leqslant
                M_{\mathbf G}(\mathcal A)+\delta_R
            \]
            \vspace{-1.0em}
        \end{block}
    \end{column}
    \begin{column}{0.6\textwidth}
        \begin{block}{Теоретическая схема законов масштабирования}
            \vspace{-0.4em}
            \begin{equation*}
                \mathcal L_m(\mathbf w_m^*)
                \lesssim
                E
                +
                \Phi(N)
                +
                \frac{
                    C\big(M_{\mathbf G}(\mathcal A)+\delta_R\big)
                }{m^{\rho}}
            \end{equation*}
            \vspace{-1.0em}
        \end{block}
        \begin{block}{Следствия}
            \begin{enumerate}
                \item Рост $N$ уменьшает ошибку выразимости $\Phi(N)$;
                \item Рост $m$ уменьшает вклад конечности выборки как $m^{-\rho}$;
                \item Коэффициент при $m^{-\rho}$ определяется архитектурой через $M_{\mathbf G}(\mathcal A)$.
            \end{enumerate}
        \end{block}
    \end{column}
\end{columns}
\begin{block}{Ключевой вывод}
    Архитектура модели влияет на достаточный размер выборки через \alert{\textbf{локальную кривизну}} и скорость \alert{\textbf{стабилизации ландшафта}} функции потерь.
\end{block}
\end{frame}

\begin{frame}{Стабилизация правдоподобия линейной регрессии в точке ОМП}
\begin{columns}[T]
    \begin{column}{0.57\textwidth}
        \vspace{-0.6em}
        \hspace*{0.1em}
        \begin{minipage}{\linewidth}
        \begin{block}{Вероятностная линейная регрессия}
            \vspace{-1.0em}
            \begin{equation*}
                \mathbf{x} \in \mathbb{R}^{n},
                \quad
                \mathbf{X}_m = [\mathbf{x}_1, \ldots, \mathbf{x}_m]^\top \in \mathbb{R}^{m \times n}
            \end{equation*}
            \begin{equation*}
                y \in \mathbb{R},
                \quad
                \mathbf{y}_m = [y_1, \ldots, y_m]^\top \in \mathbb{R}^{m}
            \end{equation*}
            \begin{equation*}
                L(\mathfrak{D}_m, \mathbf{w}) = p(\mathbf{y}_m | \mathbf{X}_m, \mathbf{w}) = \prod_{i=1}^{m} \mathcal{N}(y_i | \mathbf{w}^\top \mathbf{x}_i, \sigma^2)
            \end{equation*}
            \begin{equation*}
                \hat{\mathbf{w}}_k = \argmax_{\mathbf{w} \in \mathbb{R}^{n}} L(\mathfrak{D}_k, \mathbf{w}),
                \quad
                \mathbb{E} \hat{\mathbf{w}}_k = \mathbf{m}_k,
                \quad
                \mathbb{D} \hat{\mathbf{w}}_k = \boldsymbol{\Sigma}_k
            \end{equation*}
        \end{block}
        \end{minipage}
    \end{column}
    \hspace{0.2em}
    \begin{column}{0.4\textwidth}
        \begin{figure}
            \centering
            \includegraphics[width=\linewidth]{likelihood.png}
        \end{figure}
    \end{column}
\end{columns}
\begin{columns}[T]
    \begin{column}{0.6\textwidth}
        \hspace*{0.2em}
        \begin{minipage}{\linewidth}
        \begin{block}{$M$-достаточный размер: \textcolor{black}{$\forall k \geqslant m^*$}}
            $\alert{M(k+1)} = | \mathbb{E}_{\mathfrak{D}_{k+1}} L(\mathfrak{D}_{m}, \hat{\mathbf{w}}_{k+1}) - \mathbb{E}_{\mathfrak{D}_{k}} L(\mathfrak{D}_{m}, \hat{\mathbf{w}}_{k}) | \leqslant \varepsilon$
        \end{block}
        \end{minipage}
    \end{column}
    \vspace{0.2em}
    \begin{column}{0.4\textwidth}
        \begin{block}{$D$-достаточный размер: \textcolor{black}{$\forall k \geqslant m^*$}}
            \vspace{-2.3em}
            \begin{equation*}
                \alert{D(k)} = \mathbb{D}_{\mathfrak{D}_{k}} L(\mathfrak{D}_{m}, \hat{\mathbf{w}}_{k}) \leqslant \varepsilon
            \end{equation*}
            % $\alert{D(k)} = \mathbb{D}_{\mathfrak{D}_{k}} L(\mathfrak{D}_{m}, \hat{\mathbf{w}}_{k}) \leqslant \varepsilon$
        \end{block}
    \end{column}
\end{columns}
\vspace{-0.4em}
\begin{block}{Теорема (Киселев, 2024)}
Пусть $\| \mathbf{m}_{k+1} - \mathbf{m}_{k} \|_2 \to 0$ и $\| \boldsymbol{\Sigma}_{k+1} - \boldsymbol{\Sigma}_{k} \|_F \to 0$ при $k \to \infty$, тогда определение $M$-достаточности корректно, то есть $\forall \varepsilon > 0\ \exists m^* \in \mathbb{N}: \forall k \geqslant m^*$ $M(k) \leqslant \varepsilon$.
\end{block}
\end{frame}
